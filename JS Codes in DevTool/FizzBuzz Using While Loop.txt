var output =[];
var count = 1;

function fizzBuzz() {
while(count <= 100){
    if (count % 3 ===0){
        output.push("Fizz");
    }  else if (count % 5 ===0){
        output.push("Buzz");
    }else{
        output.push(count);
    }
count = count + 1;
   } 
    console.log(output);
}
